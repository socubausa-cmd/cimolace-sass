Code SDK — 3 lignes suffisent

/* Installation : 1 script tag */
<script src="https://app.prorascience.org/liri-sdk.js"></script>

/* Viewer public (via origin whitelist) */
const liri = LiriSDK.init({ tenant: 'isna' });
liri.viewer('#container', { session: 'SESSION_ID' });

/* Hôte (via API key — proxy serveur recommandé) */
const liri = LiriSDK.init({
  tenant: 'isna',
  apiKey: 'lk_live_xxxx',
});
liri.host('#container', { session: 'SESSION_ID' });

/* Créer une session + lancer immédiatement */
const { data: session } = await liri.createSession({
  title: 'Mon Webinar',
  session_type: 'webinar',
  scheduled_at: '2026-06-01T14:00:00Z',
});
liri.host('#container', { session: session.id });

/* Écouter les événements */
liri.on('session:started', (info) => {});
liri.on('participant:joined', (info) => {});
liri.on('session:ended', () => {});
